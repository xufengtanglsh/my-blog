class Shape:
def area(self):
raise NotImplementedError(“必须实现area方法”)

class Rectangle(Shape):
def init(self, width, height):
self.width = width
self.height = height

def area(self):
return self.width * self.height

class Circle(Shape):
def init(self, radius):
self.radius = radius

def area(self):
return 3.14 * self.radius * self.radius

def print_area(shape):
print(shape.area())

r = Rectangle(4, 5)
c = Circle(3)

print_area(r)
print_area(c)

高级SQL文①

①ROW_NUMBER() OVER
– 为所有员工按薪资排序并分配序号
SELECT
employee_name,
salary,
ROW_NUMBER() OVER (ORDER BY salary DESC) as row_num
FROM employees;

employee_name salary row_num
张三 80000 1
李四 75000 2
王五 70000 3
赵六 65000 4

– 按部门分组，每个部门内按薪资排序
SELECT
employee_name,
department,
salary,
ROW_NUMBER() OVER (
PARTITION BY department
ORDER BY salary DESC
) as dept_rank
FROM employees;

employee_name department salary dept_rank
张三 技术部 80000 1
李四 技术部 75000 2
王五 销售部 70000 1
赵六 销售部 65000 2

②RANK() OVER
　ROW_NUMBER() OVER类似， 只是他排名可以相同，相同的值分配相同的排名

③DENSE_RANK() OVER
RANK() OVER类似， 相同的值分配相同的排名,并且不会跳番

总结：
– 比较ROW_NUMBER、RANK、DENSE_RANK
SELECT
employee_name,
salary,
ROW_NUMBER() OVER (ORDER BY salary DESC) as row_num,
RANK() OVER (ORDER BY salary DESC) as rank_num,
DENSE_RANK() OVER (ORDER BY salary DESC) as dense_rank
FROM employees;

employee_name salary row_num rank_num dense_rank
张三 80000 1 1 1
李四 80000 2 1 1
王五 75000 3 3 2
赵六 70000 4 4 3
钱七 70000 5 4 3
孙八 65000 6 6 4

Java线程基础（创建、生命周期、状态）

①代码例子
public class RunnableExample {

// 实现Runnable接口
public static class MyRunnable implements Runnable {
    private String taskName;
    
    public MyRunnable(String name) {
        this.taskName = name;
    }
    
    @Override
    public void run() {
        System.out.println("任务 " + taskName + " 开始执行");
        try {
            for (int i = 0; i < 3; i++) {
                System.out.println(taskName + " 执行第 " + (i + 1) + " 次");
                Thread.sleep(500);
            }
        } catch (InterruptedException e) {
            System.out.println(taskName + " 被中断");
            Thread.currentThread().interrupt();
        }
        System.out.println("任务 " + taskName + " 执行完成");
    }
}

public static void main(String[] args) {
    // 创建Runnable对象
    MyRunnable task1 = new MyRunnable("Task-1");
    MyRunnable task2 = new MyRunnable("Task-2");
    
    // 创建线程并传入Runnable
    Thread thread1 = new Thread(task1);
    Thread thread2 = new Thread(task2);
    
    // 启动线程
    thread1.start();
    thread2.start();
}

}

②线程的生命周期
public enum ThreadState {
NEW, // 新建状态
RUNNABLE, // 可运行状态
BLOCKED, // 阻塞状态
WAITING, // 等待状态
TIMED_WAITING, // 计时等待状态
TERMINATED // 终止状态
}

③通过线程状态监听，观察线程的生命周期
public class CompleteLifecycleExample {

private static final Object lock = new Object();

public static class LifecycleThread extends Thread {
    private final String threadName;
    
    public LifecycleThread(String name) {
        this.threadName = name;
    }
    
    @Override
    public void run() {
        System.out.println(threadName + " 进入RUNNABLE状态");
        
        try {
            // 1. 进入TIMED_WAITING状态
            System.out.println(threadName + " 准备进入TIMED_WAITING状态");
            Thread.sleep(1000);
            System.out.println(threadName + " 从TIMED_WAITING状态恢复");
            
            // 2. 进入BLOCKED状态
            System.out.println(threadName + " 准备进入BLOCKED状态");
            synchronized (lock) {
                System.out.println(threadName + " 获得锁，执行同步代码");
                Thread.sleep(500);
            }
            System.out.println(threadName + " 释放锁");
            
            // 3. 进入WAITING状态
            System.out.println(threadName + " 准备进入WAITING状态");
            synchronized (lock) {
                lock.wait(2000); // 等待2秒
            }
            System.out.println(threadName + " 从WAITING状态恢复");
            
            // 4. 再次进入BLOCKED状态
            System.out.println(threadName + " 准备再次进入BLOCKED状态");
            synchronized (lock) {
                System.out.println(threadName + " 再次获得锁");
                Thread.sleep(500);
            }
            
        } catch (InterruptedException e) {
            System.out.println(threadName + " 被中断");
            Thread.currentThread().interrupt();
        }
        
        System.out.println(threadName + " 即将进入TERMINATED状态");
    }
}

public static void main(String[] args) throws InterruptedException {
    LifecycleThread thread = new LifecycleThread("LifecycleThread");
    
    // 1. NEW状态
    System.out.println("初始状态: " + thread.getState()); // NEW
    
    // 2. 启动线程，进入RUNNABLE状态
    thread.start();
    System.out.println("启动后状态: " + thread.getState()); // RUNNABLE
    
    // 3. 监控状态变化
    while (thread.isAlive()) {
        Thread.State state = thread.getState();
        System.out.println("当前状态: " + state);
        
        switch (state) {
            case NEW:
                System.out.println("线程已创建但未启动");
                break;
            case RUNNABLE:
                System.out.println("线程正在运行或等待CPU");
                break;
            case BLOCKED:
                System.out.println("线程等待获取锁");
                break;
            case WAITING:
                System.out.println("线程等待其他线程通知");
                break;
            case TIMED_WAITING:
                System.out.println("线程等待指定时间");
                break;
            case TERMINATED:
                System.out.println("线程已终止");
                break;
        }
        
        Thread.sleep(500);
    }
    
    // 4. TERMINATED状态
    System.out.println("最终状态: " + thread.getState()); // TERMINATED
}

}

sqlserverのデータ導入と導出

①導出(BATファイル)

@echo off
set SERVER=localhost
set DATABASE=DcParkBackSystem
set OUTPUT_DIR=%CD%\EXPORT
set TABLES=table1

rem 出力フォルダがなければ作成
if not exist “%OUTPUT_DIR%” mkdir “%OUTPUT_DIR%”

for %%T in (%TABLES%) do (
echo テーブルをエクスポート中: %%T
bcp “%DATABASE%.dbo.%%T” out “%OUTPUT_DIR%%%T.table” -S “%SERVER%” -T -c -t , -r \r\n
)

echo データエクスポート完了！
pause

②導入(BATファイル)

@echo off
set SERVER=localhost
set DATABASE=DcParkBackSystem
set INPUT_DIR=%CD%\IMPORT
set TABLES=table1

for %%T in (%TABLES%) do (
echo テーブルにインポート中: %%T
bcp “%DATABASE%.dbo.%%T” in “%INPUT_DIR%%%T.table” -S “%SERVER%” -T -c -t , -r \r\n
)

echo データインポート完了！
pause

注意点：　区切り交換できる： コンマ　→　¥t

特有数据类型
complex：复述类型
z1 = 3 + 4j
z2 = 1 + 2j
result = z1 * z2
print(result) # 输出: (-5+10j)

list：一般序列
    my_list = [1, 2, 3, "hello", 5.5]
tuple：	不可变序列
    my_tuple = (1, 2, 3, "hello", 5.5)
range： 数字序列
    range(5)      # 0, 1, 2, 3, 4
    range(1, 5)   # 1, 2, 3, 4
    range(1, 5, 2)  # 1, 3
dict： 映射
    person = dict(name="张三", age=25, city="北京")
    person.get("name") # 获取键值对
    person["name"] = "李四" # 修改键值对
    person.pop("age") # 删除键值对
    person.clear() # 清空字典   
set：集合不可重复
    set1 = {1, 2, 3, 4}
    set2 = {3, 4, 5, 6}
    print(set1 & set2)  # {3, 4} #交集
    print(set1 | set2)  # {1, 2, 3, 4, 5, 6} #并集
    print(set1 - set2)  # {1, 2} #差集
    
    numbers = [1, 2, 2, 3, 3, 4]
    unique_numbers = set(numbers)# 列表去重
    print(unique_numbers)  # {1, 2, 3, 4}
frozenset：不可变的集合
    frozen = frozenset([1, 2, 3]) # 创建冻结集合
    # frozen.add(4)  # 错误！不能修改

bytes：处理不可变二进制数据
    b = b'Hello'
    print(b[1:3])  # b'el'
bytearray：处理可变二进制数据
    ba = bytearray(b'Hello')
    ba[0] = 65 # 修改
    ba.append(33) # 添加   
    del ba[0] # 删除
