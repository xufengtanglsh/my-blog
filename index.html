python类声明

class Shape:
def area(self):
raise NotImplementedError(“必须实现area方法”)

class Rectangle(Shape):
def init(self, width, height):
self.width = width
self.height = height

def area(self):
return self.width * self.height

class Circle(Shape):
def init(self, radius):
self.radius = radius

def area(self):
return 3.14 * self.radius * self.radius

def print_area(shape):
print(shape.area())

r = Rectangle(4, 5)
c = Circle(3)

print_area(r)
print_area(c)

技術(dataBase)

高级SQL文①

①ROW_NUMBER() OVER
– 为所有员工按薪资排序并分配序号
SELECT
employee_name,
salary,
ROW_NUMBER() OVER (ORDER BY salary DESC) as row_num
FROM employees;

employee_name salary row_num
张三 80000 1
李四 75000 2
王五 70000 3
赵六 65000 4

– 按部门分组，每个部门内按薪资排序
SELECT
employee_name,
department,
salary,
ROW_NUMBER() OVER (
PARTITION BY department
ORDER BY salary DESC
) as dept_rank
FROM employees;

employee_name department salary dept_rank
张三 技术部 80000 1
李四 技术部 75000 2
王五 销售部 70000 1
赵六 销售部 65000 2

②RANK() OVER
　ROW_NUMBER() OVER类似， 只是他排名可以相同，相同的值分配相同的排名

③DENSE_RANK() OVER
RANK() OVER类似， 相同的值分配相同的排名,并且不会跳番

总结：
– 比较ROW_NUMBER、RANK、DENSE_RANK
SELECT
employee_name,
salary,
ROW_NUMBER() OVER (ORDER BY salary DESC) as row_num,
RANK() OVER (ORDER BY salary DESC) as rank_num,
DENSE_RANK() OVER (ORDER BY salary DESC) as dense_rank
FROM employees;

employee_name salary row_num rank_num dense_rank
张三 80000 1 1 1
李四 80000 2 1 1
王五 75000 3 3 2
赵六 70000 4 4 3
钱七 70000 5 4 3
孙八 65000 6 6 4
